# Cleaning multi-response column, preprocessing, then retraining L1-logistic (GridSearch) and RandomForest.
# Saves model results, feature importances, and plots to /mnt/data/
import pandas as pd, numpy as np, matplotlib.pyplot as plt, os, textwrap
from sklearn.model_selection import train_test_split, RepeatedStratifiedKFold, GridSearchCV
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, roc_auc_score, classification_report, confusion_matrix
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import Pipeline

file_path = "C:/Users/nirmitha/Downloads/Final minor project.xlsx"
assert os.path.exists(file_path), "Data file not found."

# Load
df = pd.read_excel(file_path)
df.columns = df.columns.str.strip()

# Target mapping
target_col = "1. Have you ever switched from one UPI app to another?"
df[target_col] = df[target_col].astype(str).str.strip().str.lower().map(lambda x: 1 if x.startswith("y") else 0)

# Multi-response column to split
multi_col = "2. If yes, what was the main reason for switching to another UPI app?"
def split_multilabel_column(df, col, delimiter=","):
    s = df[col].fillna("").astype(str).str.split(delimiter)
    s = s.apply(lambda x: [i.strip() for i in x if i.strip()!=""])
    unique = sorted({item for sublist in s for item in sublist})
    for u in unique:
        # safe column name
        colname = f"{col}__{u}"
        df[colname] = s.apply(lambda x: 1 if u in x else 0)
    return df, unique

if multi_col in df.columns:
    df, multi_labels = split_multilabel_column(df, multi_col)
    # drop original multi col
    df = df.drop(columns=[multi_col])
else:
    multi_labels = []

# Selected base features
base_features = [
    "2. Age",
    "3. Gender",
    "6. How would you classify the area you currently live in?",
    "8. How familiar are you with digital payment methods (net banking, cards, wallets, UPI)?",
    "9. Frequency of UPI usage"
]
# keep only existing ones
base_features = [c for c in base_features if c in df.columns]

# Build list of multilabel dummy columns created
multi_dummy_cols = [c for c in df.columns if c.startswith("2. If yes, what was the main reason for switching to another UPI app?__")]

# Combine features
features = base_features + multi_dummy_cols

# Drop rows with missing target or where all features missing
df = df.dropna(subset=[target_col]).copy()
# Keep rows where at least one feature is non-null or multilabel exists
df = df.reset_index(drop=True)

# Collapse rare categories for categorical base features (excluding Age if numeric)
def collapse_rare_categories(series, min_count=5):
    counts = series.value_counts(dropna=False)
    rare = counts[counts < min_count].index
    return series.replace(rare, "Other")

# Identify categorical among base_features
categorical = []
numeric = []
for c in base_features:
    if pd.api.types.is_numeric_dtype(df[c]):
        numeric.append(c)
    else:
        categorical.append(c)

for col in categorical:
    df[col] = collapse_rare_categories(df[col].astype(str), min_count=5)

# Fill numeric missing with median
for col in numeric:
    df[col] = pd.to_numeric(df[col], errors="coerce")
    df[col] = df[col].fillna(df[col].median())

# Create dummies for categorical base features
df_dummies = pd.get_dummies(df[base_features], drop_first=True)
# Combine with multilabel dummy cols
all_feature_cols = list(df_dummies.columns) + multi_dummy_cols
X = pd.concat([df_dummies, df[multi_dummy_cols].fillna(0)], axis=1)
y = df[target_col].astype(int)

# Drop any constant columns
X = X.loc[:, (X.nunique(dropna=True) > 1)]

# Train-test split stratified
X_train, X_test, y_train, y_test = train_test_split(X, y, stratify=y, test_size=0.2, random_state=42)

# Standardize (fit on train)
scaler = StandardScaler()
X_train_s = scaler.fit_transform(X_train)
X_test_s = scaler.transform(X_test)

# 1) L1 Logistic + GridSearch
param_grid = {"C": [1e-4,1e-3,1e-2,1e-1,1,10]}
cv = RepeatedStratifiedKFold(n_splits=5, n_repeats=3, random_state=42)
lr = LogisticRegression(penalty="l1", solver="liblinear", class_weight="balanced", max_iter=5000)
gs = GridSearchCV(lr, param_grid, scoring="roc_auc", cv=cv, n_jobs=-1, return_train_score=True)
gs.fit(X_train_s, y_train)
best_lr = gs.best_estimator_
# Fit best on full train (already fitted)
# Evaluate
y_train_pred_lr = best_lr.predict(X_train_s)
y_train_prob_lr = best_lr.predict_proba(X_train_s)[:,1]
y_test_pred_lr = best_lr.predict(X_test_s)
y_test_prob_lr = best_lr.predict_proba(X_test_s)[:,1]

# 2) Random Forest on processed features (no scaling needed)
rf = RandomForestClassifier(n_estimators=300, max_depth=5, random_state=42, class_weight="balanced")
rf.fit(X_train, y_train)
y_train_pred_rf = rf.predict(X_train)
y_train_prob_rf = rf.predict_proba(X_train)[:,1]
y_test_pred_rf = rf.predict(X_test)
y_test_prob_rf = rf.predict_proba(X_test)[:,1]

# Metrics and save
def summarize(name, y_tr, y_tr_pred, y_tr_prob, y_te, y_te_pred, y_te_prob):
    res = {}
    res['train_accuracy'] = accuracy_score(y_tr, y_tr_pred)
    res['train_auc'] = roc_auc_score(y_tr, y_tr_prob) if y_tr_prob is not None else None
    res['test_accuracy'] = accuracy_score(y_te, y_te_pred)
    res['test_auc'] = roc_auc_score(y_te, y_te_prob) if y_te_prob is not None else None
    res['train_report'] = classification_report(y_tr, y_tr_pred, output_dict=False)
    res['test_report'] = classification_report(y_te, y_te_pred, output_dict=False)
    res['train_cm'] = confusion_matrix(y_tr, y_tr_pred).tolist()
    res['test_cm'] = confusion_matrix(y_te, y_te_pred).tolist()
    return res

lr_res = summarize("L1-Logistic", y_train, y_train_pred_lr, y_train_prob_lr, y_test, y_test_pred_lr, y_test_prob_lr)
rf_res = summarize("RandomForest", y_train, y_train_pred_rf, y_train_prob_rf, y_test, y_test_pred_rf, y_test_prob_rf)

# Feature importances
rf_importances = pd.DataFrame({"feature": X.columns, "importance": rf.feature_importances_}).sort_values("importance", ascending=False)
lr_coefs = pd.DataFrame({"feature": X.columns, "coef": np.abs(best_lr.coef_.ravel())}).sort_values("coef", ascending=False)

# Save outputs
out_dir = "/mnt/data/cleaned_model_outputs"
os.makedirs(out_dir, exist_ok=True)
rf_importances.to_csv(os.path.join(out_dir, "rf_importances.csv"), index=False)
lr_coefs.to_csv(os.path.join(out_dir, "lr_coefs.csv"), index=False)

# Plots
plt.figure(figsize=(8,6))
plt.barh(rf_importances["feature"].head(15)[::-1], rf_importances["importance"].head(15)[::-1])
plt.title("RF top 15 importances")
plt.tight_layout()
rf_plot = os.path.join(out_dir, "rf_top15.png")

plt.close()

plt.figure(figsize=(8,6))
plt.barh(lr_coefs["feature"].head(15)[::-1], lr_coefs["coef"].head(15)[::-1])

plt.tight_layout()
lr_plot = os.path.join(out_dir, "lr_top15.png")
plt.savefig(lr_plot, dpi=150, bbox_inches="tight")
plt.close()

# Save summary json-like CSVs
pd.DataFrame([{"model":"L1-Logistic", **{k:v for k,v in lr_res.items() if k.endswith("_accuracy") or k.endswith("_auc")}},
              {"model":"RandomForest", **{k:v for k,v in rf_res.items() if k.endswith("_accuracy") or k.endswith("_auc")}}]).to_csv(os.path.join(out_dir, "model_summary.csv"), index=False)

# Print concise results
print("L1-Logistic best C:", gs.best_params_)
print("L1 Train AUC: {:.4f}, Test AUC: {:.4f}".format(lr_res['train_auc'], lr_res['test_auc']))
